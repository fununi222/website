    });
}

function filterRoutes(filter, btn) {
    document.querySelectorAll('.smbc-filter-btn').forEach(b => {
        b.classList.remove('bg-primary', 'text-background', 'active');
        b.classList.add('text-on-surface-variant');
    });
    btn.classList.add('bg-primary', 'text-background', 'active');
    btn.classList.remove('text-on-surface-variant');
    renderRoutes(filter);
