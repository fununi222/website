## 結論：自律する企業（Autonomous Enterprise）への道

2026年、私たちが目指すべきITの姿は「便利さの追求」ではなく、**「自律性の確立」**です。

1.  **インフラの自社所有化**: コアデータと計算リソースを、パブリッククラウドの「外」にも確保する。
2.  **理解の再定義**: AIを鵜呑みにせず、厳密な「要求工学」と「設計思想」を人間が主導する。
3.  **自律的防御の構築**: AIの攻撃をAIで防ぐのではなく、物理的な不変性（Immutability）という「枯れた技術」で守り抜く。

**ITはもはやコストセンターではなく、企業の意志そのものです。** 主権を取り戻した企業だけが、この非決定論的な時代を勝ち抜くことができるのです。

<script>
// 既存のスクリプトを、新しいDOM構造に合わせて再調整
(() => {
  const init = () => {
    const root = document.querySelector('.paradigm-shell');
    if (!root) return;

    const chartInstances = {};

    function navigate(targetId) {
      root.querySelectorAll('.ps-nav-item').forEach(button => {
        button.classList.toggle('active', button.dataset.target === targetId);
      });
      root.querySelectorAll('.ps-content-section').forEach(section => {
        section.classList.toggle('active', section.id === targetId);
      });
      
      if (targetId === 'ps-myth1') setTimeout(renderCloudCostChart, 100);
      if (targetId === 'ps-myth4') setTimeout(renderRansomwareChart, 100);
    }

    function renderCloudCostChart() {
      const canvas = root.querySelector('#ps-cloud-cost-chart');
      if (!canvas || typeof Chart === 'undefined') return;
      if (chartInstances.cloudCostChart) chartInstances.cloudCostChart.destroy();
      chartInstances.cloudCostChart = new Chart(canvas.getContext('2d'), {
        type: 'line',
        data: {
          labels: ['2024', '2025', '2026', '2027', '2028'],
          datasets: [
            { label: 'パブリッククラウド (累積)', data: [100, 160, 280, 480, 750], borderColor: '#fb7185', fill: true, tension: 0.4 },
            { label: 'IT主権型アーキテクチャ', data: [200, 220, 240, 260, 280], borderColor: '#6ee7b7', borderDash: [5, 5], tension: 0.1 }
          ]
        },
        options: { responsive: true, maintainAspectRatio: false }
      });
    }

    function renderRansomwareChart() {
      const canvas = root.querySelector('#ps-ransomware-chart');
      if (!canvas || typeof Chart === 'undefined') return;
      if (chartInstances.ransomwareChart) chartInstances.ransomwareChart.destroy();
      chartInstances.ransomwareChart = new Chart(canvas.getContext('2d'), {
        type: 'bar',
        data: {
          labels: ['2023', '2024', '2025'],
          datasets: [
            { type: 'line', label: 'バックアップ狙撃率 (%)', data: [80, 89, 96], borderColor: '#fb7185' },
            { type: 'bar', label: '支払い拒否率 (%)', data: [82, 84, 87], backgroundColor: '#6ee7b7' }
          ]
        },
        options: { responsive: true, maintainAspectRatio: false }
      });
    }

    root.querySelectorAll('.ps-nav-item').forEach(button => button.addEventListener('click', () => navigate(button.dataset.target)));
    navigate('ps-intro');
  };
  setTimeout(init, 500);
})();
</script>
